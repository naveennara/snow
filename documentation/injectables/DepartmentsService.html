@Injectable({
  providedIn: 'root'
})
export class DepartmentsService {
 dropdownSettings= {
    singleSelection: false,
    idField: '_id',
    textField: 'username',
    selectAllText: 'Select All',
    unSelectAllText: 'UnSelect All',
    itemsShowLimit: 3,
    allowSearchFilter: true
  };

  removeAdminsUrl(department){

  }
}
