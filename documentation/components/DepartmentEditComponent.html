import { Component, OnInit } from '@angular/core';
import { Router, ActivatedRoute } from '@angular/router';
import * as Globals from '../../../globals';
import { DepartmentsService } from '../departments.service';
import { Departments } from '../departments';
import { FormBuilder, FormGroup, Validators, FormControl } from '@angular/forms';
import { ToastrService } from 'ngx-toastr';
import { deptConstants } from '../department-constant';

@Component({
  selector: 'app-department-edit',
  templateUrl: './department-edit.component.html',
  styleUrls: ['./department-edit.component.css']
})
export class DepartmentEditComponent implements OnInit {
  departmentId: String;
  updateDepart: FormGroup;
  editDepart: Departments;
  submitted = false;
  deptKeys:any;
  constructor(private router: Router,
    private route: ActivatedRoute, private formBuilder: FormBuilder,
    private toastr: ToastrService,
    private services: DepartmentsService) { 
      this.deptKeys = deptConstants.deptkeys;
    }
  updateDepartment() {

    this.submitted = true;
    if (this.updateDepart.invalid) {
      return;
    } else {
      const url = Globals.urls.UpdateDepartment + "/" + this.departmentId;
      this.services.updateDepartment(url, JSON.stringify(this.updateDepart.value)).subscribe(
        (res: any) => {
          if (res.status === 200) {
            this.router.navigate(['mainLayout/departments'])
            this.toastr.success(res.message);
          }
          else {
            this.toastr.info(res.message);
          }
        }
      );

    }
  }


  editDepartment() {
    const url = Globals.urls.editDepartment + '/' + this.departmentId;
    this.services.editDepartment(url).subscribe(
      (res: any) => {
       
          if (res.status === 200) {
            this.editDepart = res.data;         
            this.updateDepart.patchValue(this.editDepart);
          }
          else if (res.status === 204) {
          }
      
      });
  }
 

  get f() { return this.updateDepart.controls; }
  
  ngOnInit() {
    this.route.params.subscribe(params => {
      this.departmentId = params['departmentId'];

    });
    this.editDepartment();
    this.updateDepart = this.formBuilder.group({
      name: ['', Validators.required],
      email: ['',[Validators.required, Validators.email]],
      description: ['']
    });
  }

}
